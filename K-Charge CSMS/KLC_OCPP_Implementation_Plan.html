1.2. OCPP Gateway Component

OCPP Gateway là một background service chạy trong cùng process với ASP.NET Core Host, lắng nghe WebSocket connections từ các Charge Point. Mỗi trạm sạc duy trì một persistent WebSocket connection.

WebSocket URL Pattern: ws://csms.odcall.com/ocpp/{chargePointId}
ChargePointId Format: KLC-{CITY}-{SEQ} (ví dụ: KLC-HCM-001, KLC-HN-015)
Protocol: OCPP 1.6J (JSON over WebSocket), sub-protocol: ocpp1.6

2. WebSocket Server & OCPP Gateway

2.1. WebSocket Middleware

ASP.NET Core hỗ trợ WebSocket natively. OCPP Gateway sẽ sử dụng middleware pattern để accept và quản lý connections.

// Program.cs - KLC.HttpApi.Host
var app = builder.Build();

// Enable WebSocket support
app.UseWebSockets(new WebSocketOptions
{
    KeepAliveInterval = TimeSpan.FromSeconds(30),
    AllowedOrigins = { "*" }  // Restrict in production
});

// OCPP WebSocket endpoint - MUST be before MapControllers
app.UseOcppWebSocket("/ocpp");  // Handles ws://host/ocpp/{chargePointId}

app.MapControllers();

2.3. OCPP Message Format (JSON)

OCPP 1.6J defines 3 message types transported as JSON arrays over WebSocket:

TypeIDFormatDirection
CALL2[2, "uniqueId", "Action", {payload}]Request (either direction)
CALLRESULT3[3, "uniqueId", {payload}]Response to CALL
CALLERROR4[4, "uniqueId", "errorCode", "errorDesc", {details}]Error response
UniqueId: Chuỗi tối đa 36 ký tự, dùng để match request/response. Recommend sử dụng GUID. Charge Point gửi CALL, CSMS reply CALLRESULT/CALLERROR với cùng UniqueId.

2.4. Message Router

public class OcppMessageRouter
{
    private readonly IServiceProvider _serviceProvider;
    private readonly ILogger<OcppMessageRouter> _logger;

    // Map Action name → Handler type
    private static readonly Dictionary<string, Type> _handlers = new()
    {
        ["BootNotification"]      = typeof(BootNotificationHandler),
        ["Heartbeat"]             = typeof(HeartbeatHandler),
        ["StatusNotification"]    = typeof(StatusNotificationHandler),
        ["Authorize"]             = typeof(AuthorizeHandler),
        ["StartTransaction"]      = typeof(StartTransactionHandler),
        ["StopTransaction"]       = typeof(StopTransactionHandler),
        ["MeterValues"]           = typeof(MeterValuesHandler),
        ["DataTransfer"]          = typeof(DataTransferHandler),
        ["DiagnosticsStatusNotification"] = typeof(DiagnosticsStatusNotificationHandler),
        ["FirmwareStatusNotification"]    = typeof(FirmwareStatusNotificationHandler),
    };

    public async Task<JArray> RouteMessageAsync(
        string chargePointId, JArray message, CancellationToken ct)
    {
        int messageTypeId = message[0].Value<int>();

        switch (messageTypeId)
        {
            case 2: // CALL from Charge Point
                string uniqueId = message[1].Value<string>();
                string action = message[2].Value<string>();
                JObject payload = message[3] as JObject ?? new JObject();

                if (!_handlers.TryGetValue(action, out var handlerType))
                {
                    // Return CALLERROR: NotImplemented
                    return new JArray(4, uniqueId, "NotImplemented",
                        $"Action '{action}' not supported", new JObject());
                }

                using var scope = _serviceProvider.CreateScope();
                var handler = (IOcppMessageHandler)scope.ServiceProvider
                    .GetRequiredService(handlerType);

                var result = await handler.HandleAsync(chargePointId, payload, ct);
                return new JArray(3, uniqueId, result); // CALLRESULT

            case 3: // CALLRESULT (response to our request)
            case 4: // CALLERROR (error response to our request)
                // Route to pending request completion
                HandleResponseMessage(chargePointId, message);
                return null; // No response needed

            default:
                _logger.LogWarning("Unknown message type {Type} from {CP}",
                    messageTypeId, chargePointId);
                return null;
        }
    }
}

2.6. Sending Commands to Charge Point (CSMS → CP)

public class OcppMessageDispatcher : IOcppMessageDispatcher
{
    private readonly OcppConnectionManager _connectionManager;

    public async Task<JObject> SendCommandAsync(
        string chargePointId, string action, JObject payload,
        TimeSpan? timeout = null, CancellationToken ct = default)
    {
        var conn = _connectionManager.GetConnection(chargePointId)
            ?? throw new ChargePointNotConnectedException(chargePointId);

        string uniqueId = Guid.NewGuid().ToString("N")[..36];
        var tcs = new TaskCompletionSource<JArray>();

        conn.PendingRequests[uniqueId] = tcs;

        try
        {
            // Send CALL message
            var message = new JArray(2, uniqueId, action, payload);
            var bytes = Encoding.UTF8.GetBytes(message.ToString(Formatting.None));
            await conn.WebSocket.SendAsync(bytes, WebSocketMessageType.Text,
                true, ct);

            // Wait for response with timeout (default 30s)
            using var cts = CancellationTokenSource
                .CreateLinkedTokenSource(ct);
            cts.CancelAfter(timeout ?? TimeSpan.FromSeconds(30));

            var response = await tcs.Task.WaitAsync(cts.Token);

            int responseType = response[0].Value<int>();
            if (responseType == 3) // CALLRESULT
                return response[2] as JObject;
            else // CALLERROR
                throw new OcppCallErrorException(
                    response[2].Value<string>(),
                    response[3].Value<string>());
        }
        finally
        {
            conn.PendingRequests.TryRemove(uniqueId, out _);
        }
    }
}

3. OCPP Message Handlers (Chi Tiết)

3.1. Charge Point → CSMS (CP-Initiated)

Các message được Charge Point gửi đến CSMS. Đây là các handler cần implement ở phía server.

3.1.1. BootNotification

Gửi khi Charge Point khởi động hoặc kết nối lại. CSMS quyết định accept/reject/pending.

FieldTypeRequiredDescription
chargePointVendorCiString20Yes"Chargecore" hoặc "JUHANG"
chargePointModelCiString20Yes"AC003+", "EVMS-60", etc.
chargePointSerialNumberCiString25NoSerial number
chargeBoxSerialNumberCiString25NoCharge box serial
firmwareVersionCiString50NoFirmware version string
iccidCiString20NoSIM ICCID (4G models)
imsiCiString20NoSIM IMSI
meterTypeCiString25NoMeter type
meterSerialNumberCiString25NoMeter serial

Response:

FieldTypeDescription
statusRegistrationStatusAccepted / Pending / Rejected
currentTimedateTimeServer UTC time (for charger clock sync)
intervalintegerHeartbeat interval in seconds (recommend: 300)
public class BootNotificationHandler : IOcppMessageHandler
{
    private readonly IChargingStationRepository _stationRepo;
    private readonly IUnitOfWorkManager _uowManager;
    private readonly ILogger<BootNotificationHandler> _logger;

    public async Task<JObject> HandleAsync(
        string chargePointId, JObject payload, CancellationToken ct)
    {
        var vendor = payload.Value<string>("chargePointVendor");
        var model = payload.Value<string>("chargePointModel");
        var serial = payload.Value<string>("chargePointSerialNumber");
        var firmware = payload.Value<string>("firmwareVersion");

        using var uow = _uowManager.Begin();

        var station = await _stationRepo.FindByChargePointIdAsync(chargePointId);

        RegistrationStatus status;
        if (station == null)
        {
            // Unknown charger → Pending (requires admin registration first)
            _logger.LogWarning(
                "Unknown ChargePoint {Id} attempted BootNotification. " +
                "Vendor={Vendor}, Model={Model}",
                chargePointId, vendor, model);
            status = RegistrationStatus.Pending;
        }
        else
        {
            // Known charger → Update info and accept
            station.UpdateBootInfo(vendor, model, serial, firmware,
                payload.Value<string>("iccid"),
                payload.Value<string>("imsi"));
            station.SetOnline();

            await _stationRepo.UpdateAsync(station);
            status = RegistrationStatus.Accepted;
        }

        await uow.CompleteAsync();

        return new JObject
        {
            ["status"] = status.ToString(),
            ["currentTime"] = DateTime.UtcNow.ToString("o"),
            ["interval"] = 300  // Heartbeat every 5 minutes
        };
    }
}
Quan trọng: Nếu CSMS trả về Pending hoặc Rejected, Charge Point sẽ gửi lại BootNotification sau interval giây. Trong thời gian này, Charge Point KHÔNG nên gửi bất kỳ message nào khác (trừ BootNotification). Admin cần register trạm trong hệ thống trước khi charger có thể hoạt động.

3.1.3. StatusNotification

Connector status thay đổi. Đây là message quan trọng nhất cho monitoring real-time.

FieldTypeRequiredDescription
connectorIdintegerYes0 = charger overall, >=1 = specific connector
errorCodeChargePointErrorCodeYesNoError, ConnectorLockFailure, GroundFailure, ...
statusChargePointStatusYesAvailable, Preparing, Charging, Faulted, ...
timestampdateTimeNoTime of status change
infoCiString50NoAdditional info
vendorIdCiString255NoVendor-specific ID
vendorErrorCodeCiString50NoVendor-specific error code
public class StatusNotificationHandler : IOcppMessageHandler
{
    private readonly IChargingStationRepository _stationRepo;
    private readonly IDistributedEventBus _eventBus;

    public async Task<JObject> HandleAsync(
        string chargePointId, JObject payload, CancellationToken ct)
    {
        int connectorId = payload.Value<int>("connectorId");
        var status = Enum.Parse<ChargePointStatus>(
            payload.Value<string>("status"));
        var errorCode = Enum.Parse<ChargePointErrorCode>(
            payload.Value<string>("errorCode"));
        var timestamp = payload.Value<DateTime?>("timestamp") ?? DateTime.UtcNow;

        var station = await _stationRepo
            .FindByChargePointIdAsync(chargePointId);
        if (station == null) return new JObject(); // Empty response per spec

        if (connectorId == 0)
        {
            // Overall charger status
            station.UpdateStatus(status, errorCode, timestamp);
        }
        else
        {
            // Specific connector status
            var connector = station.Connectors
                .FirstOrDefault(c => c.ConnectorId == connectorId);
            if (connector != null)
            {
                connector.UpdateStatus(status, errorCode, timestamp);
            }
        }

        // Log fault if error
        if (errorCode != ChargePointErrorCode.NoError)
        {
            station.AddFault(connectorId, errorCode,
                payload.Value<string>("info"),
                payload.Value<string>("vendorErrorCode"),
                timestamp);
        }

        await _stationRepo.UpdateAsync(station);

        // Publish real-time event for Admin Portal (SignalR)
        await _eventBus.PublishAsync(new ConnectorStatusChangedEto
        {
            ChargePointId = chargePointId,
            ConnectorId = connectorId,
            Status = status,
            ErrorCode = errorCode,
            Timestamp = timestamp
        });

        return new JObject(); // Empty response per OCPP spec
    }
}

3.1.4. Authorize

Validate RFID tag hoặc ID token trước khi cho phép sạc.

public class AuthorizeHandler : IOcppMessageHandler
{
    private readonly IIdTagRepository _idTagRepo;

    public async Task<JObject> HandleAsync(
        string chargePointId, JObject payload, CancellationToken ct)
    {
        string idTag = payload.Value<string>("idTag");

        var tag = await _idTagRepo.FindAsync(idTag);

        var idTagInfo = new JObject();
        if (tag == null)
        {
            idTagInfo["status"] = "Invalid";
        }
        else if (tag.IsBlocked)
        {
            idTagInfo["status"] = "Blocked";
        }
        else if (tag.ExpiryDate.HasValue && tag.ExpiryDate < DateTime.UtcNow)
        {
            idTagInfo["status"] = "Expired";
        }
        else
        {
            idTagInfo["status"] = "Accepted";
            if (tag.ExpiryDate.HasValue)
                idTagInfo["expiryDate"] = tag.ExpiryDate.Value.ToString("o");
            if (!string.IsNullOrEmpty(tag.ParentIdTag))
                idTagInfo["parentIdTag"] = tag.ParentIdTag;
        }

        return new JObject { ["idTagInfo"] = idTagInfo };
    }
}

3.1.5. StartTransaction

Bắt đầu một charging session. CSMS phải trả về transactionId.

public class StartTransactionHandler : IOcppMessageHandler
{
    private readonly IChargingSessionRepository _sessionRepo;
    private readonly IIdTagRepository _idTagRepo;
    private readonly IChargingStationRepository _stationRepo;

    public async Task<JObject> HandleAsync(
        string chargePointId, JObject payload, CancellationToken ct)
    {
        int connectorId = payload.Value<int>("connectorId");
        string idTag = payload.Value<string>("idTag");
        int meterStart = payload.Value<int>("meterStart"); // Wh
        var timestamp = payload.Value<DateTime>("timestamp");
        int? reservationId = payload.Value<int?>("reservationId");

        // Validate idTag
        var tag = await _idTagRepo.FindAsync(idTag);
        var authStatus = ValidateTag(tag);

        if (authStatus != AuthorizationStatus.Accepted)
        {
            return new JObject
            {
                ["transactionId"] = 0,
                ["idTagInfo"] = new JObject { ["status"] = authStatus.ToString() }
            };
        }

        // Create charging session
        var session = new ChargingSession(
            GuidGenerator.Create(),
            chargePointId,
            connectorId,
            idTag,
            meterStart,
            timestamp
        );

        await _sessionRepo.InsertAsync(session);

        return new JObject
        {
            ["transactionId"] = session.TransactionId, // Auto-increment int
            ["idTagInfo"] = new JObject
            {
                ["status"] = "Accepted",
                ["expiryDate"] = tag?.ExpiryDate?.ToString("o")
            }
        };
    }
}
transactionId: OCPP spec yêu cầu integer. Sử dụng auto-increment column trong PostgreSQL. Đây là ID dùng trong giao tiếp OCPP, khác với Guid Id của entity trong ABP.

3.1.6. StopTransaction

public class StopTransactionHandler : IOcppMessageHandler
{
    private readonly IChargingSessionRepository _sessionRepo;
    private readonly IDistributedEventBus _eventBus;

    public async Task<JObject> HandleAsync(
        string chargePointId, JObject payload, CancellationToken ct)
    {
        int transactionId = payload.Value<int>("transactionId");
        int meterStop = payload.Value<int>("meterStop"); // Wh
        var timestamp = payload.Value<DateTime>("timestamp");
        var reason = payload.Value<string>("reason") ?? "Local";

        var session = await _sessionRepo
            .FindByTransactionIdAsync(transactionId);

        if (session == null)
        {
            // Unknown transaction - still accept per spec
            return new JObject();
        }

        // Calculate energy consumed
        session.Stop(meterStop, timestamp, reason);

        // Process transactionData (final meter values)
        var transactionData = payload["transactionData"] as JArray;
        if (transactionData != null)
        {
            foreach (var meterValue in transactionData)
            {
                session.AddMeterValue(ParseMeterValue(meterValue));
            }
        }

        await _sessionRepo.UpdateAsync(session);

        // Trigger billing calculation
        await _eventBus.PublishAsync(new ChargingSessionCompletedEto
        {
            SessionId = session.Id,
            TransactionId = transactionId,
            ChargePointId = chargePointId,
            EnergyConsumedWh = meterStop - session.MeterStartWh,
            Duration = timestamp - session.StartTimestamp
        });

        return new JObject
        {
            ["idTagInfo"] = new JObject { ["status"] = "Accepted" }
        };
    }
}

3.1.7. MeterValues

Dữ liệu đo lường định kỳ (kWh, voltage, current, power, SoC).

public class MeterValuesHandler : IOcppMessageHandler
{
    private readonly IChargingSessionRepository _sessionRepo;
    private readonly IDistributedEventBus _eventBus;

    public async Task<JObject> HandleAsync(
        string chargePointId, JObject payload, CancellationToken ct)
    {
        int connectorId = payload.Value<int>("connectorId");
        int? transactionId = payload.Value<int?>("transactionId");
        var meterValues = payload["meterValue"] as JArray;

        if (transactionId.HasValue && meterValues != null)
        {
            var session = await _sessionRepo
                .FindByTransactionIdAsync(transactionId.Value);

            if (session != null)
            {
                foreach (var mv in meterValues)
                {
                    var timestamp = mv.Value<DateTime>("timestamp");
                    var sampledValues = mv["sampledValue"] as JArray;

                    foreach (var sv in sampledValues)
                    {
                        session.AddMeterValue(new MeterValue(
                            GuidGenerator.Create(),
                            session.Id,
                            timestamp,
                            sv.Value<string>("value"),
                            sv.Value<string>("measurand")
                                ?? "Energy.Active.Import.Register",
                            sv.Value<string>("unit") ?? "Wh",
                            sv.Value<string>("context") ?? "Sample.Periodic",
                            sv.Value<string>("format") ?? "Raw",
                            sv.Value<string>("location") ?? "Outlet",
                            sv.Value<string>("phase")
                        ));
                    }
                }

                await _sessionRepo.UpdateAsync(session);

                // Publish for real-time dashboard updates
                await _eventBus.PublishAsync(new MeterValueReceivedEto
                {
                    ChargePointId = chargePointId,
                    ConnectorId = connectorId,
                    TransactionId = transactionId.Value,
                    MeterValues = ParseMeterValues(meterValues)
                });
            }
        }

5.2. WebSocket Handshake Validation

// In OcppWebSocketMiddleware
private async Task HandleWebSocketConnection(HttpContext context)
{
    // Extract ChargePointId from URL path
    string chargePointId = ExtractChargePointId(context.Request.Path);

    // Validate ChargePointId is registered
    var station = await _stationRepo.FindByChargePointIdAsync(chargePointId);
    if (station == null)
    {
        _logger.LogWarning("Rejected unknown ChargePoint: {Id}", chargePointId);
        context.Response.StatusCode = 404;
        return;
    }

    // Validate sub-protocol
    if (!context.WebSockets.WebSocketRequestedProtocols.Contains("ocpp1.6"))
    {
        context.Response.StatusCode = 400;
        return;
    }

    // Optional: Basic Auth validation
    if (!ValidateCredentials(context, station))
    {
        context.Response.StatusCode = 401;
        return;
    }

    // Accept WebSocket with OCPP sub-protocol
    var ws = await context.WebSockets.AcceptWebSocketAsync("ocpp1.6");

    // Register connection and start message loop
    var connection = new OcppConnection
    {
        ChargePointId = chargePointId,
        WebSocket = ws,
        ConnectedAt = DateTime.UtcNow
    };

    _connectionManager.TryAdd(chargePointId, connection);
    await ProcessMessagesAsync(connection);
}

7. Offline & Reconnection Handling

7.1. Charger Goes Offline

Khi mất kết nối WebSocket:

  1. CSMS detect qua missing heartbeat (timeout = HeartbeatInterval × 2 + 30s buffer)
  2. Mark station IsOnline = false, status = Unavailable
  3. Background job kiểm tra heartbeat timeout mỗi 60 giây
  4. Charger tự động reconnect và gửi BootNotification
  5. Nếu có active session khi offline, charger queue StopTransaction và gửi khi reconnect

7.2. Offline Transaction Handling

OCPP Offline Behavior: Khi charger mất kết nối, nó tiếp tục cho phép sạc (nếu có Local Authorization List). Khi reconnect, charger gửi tất cả queued StartTransaction và StopTransaction messages. CSMS phải xử lý idempotent - check transactionId để tránh duplicate.

7.3. Heartbeat Monitor Background Job

public class HeartbeatMonitorWorker : AsyncPeriodicBackgroundWorkerBase
{
    private readonly OcppConnectionManager _connectionManager;
    private readonly IChargingStationRepository _stationRepo;

    protected override async Task DoWorkAsync(
        PeriodicBackgroundWorkerContext context)
    {
        var timeout = TimeSpan.FromMinutes(12); // 2 × 300s + 120s buffer
        var now = DateTime.UtcNow;

        foreach (var cp in _connectionManager.GetConnectedChargePoints())
        {
            var conn = _connectionManager.GetConnection(cp);
            if (conn != null && now - conn.LastHeartbeat > timeout)
            {
                _logger.LogWarning("ChargePoint {Id} heartbeat timeout", cp);

                // Close stale WebSocket
                await conn.WebSocket.CloseAsync(
                    WebSocketCloseStatus.NormalClosure,
                    "Heartbeat timeout", CancellationToken.None);
                _connectionManager.TryRemove(cp, out _);

                // Update DB
                var station = await _stationRepo.FindByChargePointIdAsync(cp);
                if (station != null)
                {
                    station.SetOffline();
                    await _stationRepo.UpdateAsync(station);
                }
            }
        }
    }
}

"""Simple OCPP 1.6J Charge Point simulator for testing KLC CSMS"""
import asyncio, json, uuid
import websockets

CSMS_URL = "ws://localhost:8081/ocpp/KLC-HCM-TEST01"

async def simulate():
    async with websockets.connect(CSMS_URL, subprotocols=["ocpp1.6"]) as ws:
        # 1. BootNotification
        boot = [2, str(uuid.uuid4())[:36], "BootNotification", {
            "chargePointVendor": "Chargecore",
            "chargePointModel": "AC003+",
            "firmwareVersion": "1.2.0"
        }]
        await ws.send(json.dumps(boot))
        resp = json.loads(await ws.recv())
        print(f"Boot: {resp}")

        # 2. StatusNotification - Connector 1 Available
        status = [2, str(uuid.uuid4())[:36], "StatusNotification", {
            "connectorId": 1, "errorCode": "NoError", "status": "Available"
        }]
